    namespace: default

    schedule: "0 2 * * *"
    timezone: {{ .Values.scheduleDefaults.timezone }}
    jobTemplate:
      spec:
        source:
          provider: kubevirt
          namespace: default
          vmName: my-vm

          type: s3

  EOF
